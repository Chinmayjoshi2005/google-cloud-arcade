### Run the following Commands in CloudShell


```bash
cat <<EOF > lifecycle.json
{
  "lifecycle": {
    "rule": [
      {
        "action": {"type": "SetStorageClass", "storageClass": "STANDARD"},
        "condition": {
          "age": 30,
          "matchesPrefix": ["projects/active/"]
        }
      },
      {
        "action": {"type": "SetStorageClass", "storageClass": "NEARLINE"},
        "condition": {
          "age": 90,
          "matchesPrefix": ["archive/"]
        }
      },
      {
        "action": {"type": "SetStorageClass", "storageClass": "COLDLINE"},
        "condition": {
          "age": 180,
          "matchesPrefix": ["archive/"]
        }
      },
      {
        "action": {"type": "Delete"},
        "condition": {
          "age": 7,
          "matchesPrefix": ["processing/temp_logs/"]
        }
      }
    ]
  }
}
EOF
```
```bash
gcloud storage buckets update gs://[REPLACE WITH YOUR BUCKET_NAME] --lifecycle-file=lifecycle.json
```
```bash
gcloud storage buckets describe gs://[REPLACE WITH YOUR BUCKET_NAME] --format="json(lifecycle)"
```
